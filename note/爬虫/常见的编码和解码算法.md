# python
# pip install pyDes
import binascii
from pyDes import des,CBC,PAD_PKCSS
def des_encrypt(secret_key,s):
    iv = secret_key
    k = des(secret_key,CBC,iv,pad=None,padmode=PAD_PKCSS)
    en = k.encrypt(s,padmode=PAD_PKCSS)
    return binascii.b2a_hex(en)
def des_decrypt(secret_key,s):
    iv = secret_key
    k = des(secret_key,CBC,pad=None,padmode=PAD_PKCSS)
    de = k.decrypt(binascii.a2b_hex(s),padmode=PAD_PKCSS)
    return de
secret_str = des_encrypt('9999','fdsa')
print(secret_str.decode())
clear_str = des_decrypt('0000',secret_str)
print(clear_str.decode())
